352template <SupportedFigure T>
353T* GeometryStorage::get(ID id) noexcept {
354 auto it = m_index.find(id);
355 if (it == m_index.end() || it->second.type != typeToEnum<T>()) {
356 return nullptr;
357 }
358 const std::uint32_t s = it->second.slot;
359 if constexpr (std::same_as<T, Point2D>) {
360 return m_pointSlots[s].get();

362 return m_lineSlots[s].get();
363 } else if constexpr (std::same_as<T, Circle2D>) {
364 return m_circleSlots[s].get();
365 } else {
366 return m_arcSlots[s].get();
367 }

392template <typename T>
393void GeometryStorage::eraseCachedById(
394 ID id,
395 std::vector<FigureRef<T>>& cache,
396 std::unordered_map<ID, std::size_t>& posMap
397) noexcept {
398 auto it = posMap.find(id);
399 if (it == posMap.end()) {
400 return;
401 }
402 const std::size_t idx = it->second;
403 const std::size_t lastIdx = cache.size() - 1;
404 if (idx != lastIdx) {
405 cache[idx] = cache[lastIdx];
406 posMap[cache[idx].id] = idx;
407 }
408 cache.pop_back();
409 posMap.erase(it);
410}
