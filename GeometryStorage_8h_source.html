406template <SupportedFigure T>
407T* GeometryStorage::get(ID id) const {
408 auto it = m_index.find(id);
409 if (it == m_index.end()) {
410 throw std::runtime_error("Object with given ID not found");
411 }
412
413 const FigureEntry& entry = it->second;
414 if (entry.type != typeToEnum<T>()) {
415 throw std::runtime_error("Requested type does not match actual type");
416 }
417
418 return static_cast<T*>(entry.ptr);
419}
