136 bool validate() const {
137 switch (type) {
138 case FigureType::ET_POINT2D:
139 if (coords.size() == 2) {
140 break;
141 }
142 if (!x.has_value() || !y.has_value()) {
143 throw std::invalid_argument("Point requires x and y coordinates");
144 }
145 break;
146 case FigureType::ET_LINE:
147 if (coords.size() == 4) {
148 break;
149 }
150 if (pointIds.size() != 2) {
151 throw std::invalid_argument("Line requires coordinates or exactly 2 point IDs");
152 }
153 break;
154 case FigureType::ET_CIRCLE:
155 if (coords.size() != 2 && pointIds.size() != 1) {
156 throw std::invalid_argument("Circle requires center coordinates or exactly 1 center point ID");
157 }
158 if (!radius.has_value() || radius.value() <= 0) {
159 throw std::invalid_argument("Circle requires positive radius");
160 }
161 break;
162 case FigureType::ET_ARC:
163 if (coords.size() == 6) {
164 break;
165 }
166 if (pointIds.size() != 3) {
167 throw std::invalid_argument("Arc requires coordinates or exactly 3 point IDs");
168 }
169 break;
170 }
171 return true;
172 }
