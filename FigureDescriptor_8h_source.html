135 bool validate() const {
136 switch (type) {
137 case FigureType::ET_POINT2D:
138 if (coords.size() == 2) {
139 break;
140 }
141 if (!x.has_value() || !y.has_value()) {
142 throw std::invalid_argument("Point requires x and y coordinates");
143 }
144 break;
145 case FigureType::ET_LINE:
146 if (coords.size() == 4) {
147 break;
148 }
149 if (pointIds.size() != 2) {
150 throw std::invalid_argument("Line requires coordinates or exactly 2 point IDs");
151 }
152 break;
153 case FigureType::ET_CIRCLE:
154 if (coords.size() != 2 && pointIds.size() != 1) {
155 throw std::invalid_argument("Circle requires center coordinates or exactly 1 center point ID");
156 }
157 if (!radius.has_value() || radius.value() <= 0) {
158 throw std::invalid_argument("Circle requires positive radius");
159 }
160 break;
161 case FigureType::ET_ARC:
162 if (coords.size() == 6) {
163 break;
164 }
165 if (pointIds.size() != 3) {
166 throw std::invalid_argument("Arc requires coordinates or exactly 3 point IDs");
