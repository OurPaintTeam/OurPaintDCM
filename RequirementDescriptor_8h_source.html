146 case RequirementType::ET_ARCCENTERONPERPENDICULAR:
147 if (objectIds.size() != 1) {
148 throw std::invalid_argument("Requirement type requires exactly 1 object ID");
149 }
150 break;
151 }

153 // Check that param is provided for types that need it
154 switch (type) {

156 case RequirementType::ET_POINTPOINTDIST:
157 case RequirementType::ET_LINECIRCLEDIST:
158 case RequirementType::ET_LINELINEANGLE:
159 if (!param.has_value()) {
160 throw std::invalid_argument("Requirement type requires a parameter value");
161 }
162 break;
